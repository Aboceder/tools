    <title>文本反转器</title>

    <!-- 引入 Tailwind CSS -->

    <script src="https://cdn.tailwindcss.com"></script>

    <style>

        /* 自定义 Tailwind 配置，确保 Inter 字体可用 */

        @import url('https://fonts.googleapis.com/css2?family=Inter:wght@400;500;600;700&display=swap');

        }

    </style>

</head>

<body class="bg-gradient-to-br from-blue-50 to-indigo-100 min-h-screen flex items-center justify-center p-4">

    <div class="bg-white p-8 rounded-2xl shadow-xl w-full max-w-2xl transform transition-all duration-300 hover:shadow-2xl">

        <h1 class="text-4xl font-extrabold text-center text-gray-800 mb-8 tracking-tight">

            文本反转器

        </h1>


        <div class="mb-6">

            <label for="inputText" class="block text-lg font-semibold text-gray-700 mb-2">

                输入您的文字:

            </label>

            <textarea

                id="inputText"

                class="w-full p-4 border-2 border-gray-300 rounded-lg focus:outline-none focus:ring-4 focus:ring-blue-300 transition duration-200 resize-y min-h-[120px] text-gray-800 text-base"

                placeholder="在此输入您想反转的文字..."

            ></textarea>

        </div>


        <button

            id="reverseButton"

            class="w-full bg-gradient-to-r from-blue-500 to-indigo-600 text-white font-bold py-3 px-6 rounded-xl shadow-lg hover:from-blue-600 hover:to-indigo-700 transition duration-300 transform hover:scale-105 focus:outline-none focus:ring-4 focus:ring-blue-400 focus:ring-opacity-75"

        >

            反转文字

        </button>


        <div class="mt-8">

            <label for="outputText" class="block text-lg font-semibold text-gray-700 mb-2">

                反转后的文字:

            </label>

            <textarea

                id="outputText"

                class="w-full p-4 border-2 border-gray-200 bg-gray-50 rounded-lg cursor-not-allowed text-gray-700 text-base min-h-[120px] resize-y"

                readonly

                placeholder="反转后的文字将显示在这里..."

            ></textarea>

        </div>


        <!-- 消息框用于显示复制成功等提示 -->

        <div id="messageBox" class="fixed bottom-6 right-6 bg-green-500 text-white px-5 py-3 rounded-xl shadow-lg transform translate-y-full opacity-0 transition-all duration-300 ease-in-out">

            文字已复制到剪贴板！

        </div>


        <button

            id="copyButton"

            class="w-full bg-gray-200 text-gray-800 font-bold py-3 px-6 rounded-xl shadow-md hover:bg-gray-300 transition duration-300 transform hover:scale-105 mt-4 focus:outline-none focus:ring-4 focus:ring-gray-300 focus:ring-opacity-75"

        >

            复制反转后的文字

        </button>

    </div>


    <script>

        // 获取 DOM 元素

        const inputText = document.getElementById('inputText');

        const reverseButton = document.getElementById('reverseButton');

        const outputText = document.getElementById('outputText');

        const copyButton = document.getElementById('copyButton');

        const messageBox = document.getElementById('messageBox');


        // 反转文字的函数

        function reverseText() {

            const originalText = inputText.value;

            // 将字符串转换为数组，反转数组，再将数组连接回字符串

            const reversed = originalText.split('').reverse().join('');

            outputText.value = reversed;

        }


        // 复制反转文字到剪贴板的函数

        function copyToClipboard() {

            // 选择输出文本框中的内容

            outputText.select();

            // 尝试执行复制命令

            try {

                const successful = document.execCommand('copy');

                if (successful) {

                    showMessage("文字已复制到剪贴板！");

                } else {

                    showMessage("复制失败，请手动复制。", true);

                }

            } catch (err) {

                showMessage("复制失败，您的浏览器可能不支持此操作。", true);

                console.error('复制命令失败', err);

            }

            // 取消选择

            window.getSelection().removeAllRanges();

        }


        // 显示消息框的函数

        function showMessage(message, isError = false) {

            messageBox.textContent = message;

            if (isError) {

                messageBox.classList.remove('bg-green-500');

                messageBox.classList.add('bg-red-500');

            } else {

                messageBox.classList.remove('bg-red-500');

                messageBox.classList.add('bg-green-500');

            }


            // 显示消息框

            messageBox.classList.remove('translate-y-full', 'opacity-0');

            messageBox.classList.add('translate-y-0', 'opacity-100');


            // 2秒后隐藏消息框

            setTimeout(() => {

                messageBox.classList.remove('translate-y-0', 'opacity-100');

                messageBox.classList.add('translate-y-full', 'opacity-0');

            }, 2000);

        }


        // 添加事件监听器

        reverseButton.addEventListener('click', reverseText);

        copyButton.addEventListener('click', copyToClipboard);


        // 初始化时也反转一次（如果输入框有默认内容）

        reverseText();

    </script>

</body>
